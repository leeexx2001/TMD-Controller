```python
    return p


def read_file_lines(
    filepath: Union[Path, str],
    encoding: str = "utf-8",
    strip_lines: bool = True,
    skip_empty: bool = True,
) -> List[str]:
    """读取文件并返回行列表"""
    p = Path(filepath) if isinstance(filepath, str) else filepath

    with open(p, "r", encoding=encoding) as f:
        lines = f.readlines()

    result = []
    for line in lines:
        if strip_lines:
            line = line.strip()
        if skip_empty and not line:
            continue
        result.append(line)

    return result


def get_file_size(filepath: Union[Path, str]) -> int:
    """获取文件大小（字节）"""
    p = Path(filepath) if isinstance(filepath, str) else filepath

    try:
        return p.stat().st_size
    except (FileNotFoundError, OSError):
        return 0
```

```python
def format_file_size(size_bytes: int, precision: int = 2) -> str:
    """格式化文件大小为人类可读字符串"""
    if size_bytes < 0:
        return "无效大小"

    if size_bytes == 0:
        return "0 B"

    units = ["B", "KB", "MB", "GB", "TB", "PB"]
    unit_index = 0
    size = float(size_bytes)

    while size >= 1024 and unit_index < len(units) - 1:
        size /= 1024
        unit_index += 1

    return f"{size:.{precision}f} {units[unit_index]}"


def format_number(
    number: Union[int, float], thousand_separator: str = ",", precision: int = 0
) -> str:
    """格式化数字为带千分位的字符串"""
    if isinstance(number, float) and precision > 0:
        formatted = f"{number:,.{precision}f}"
        if thousand_separator != ",":
            formatted = formatted.replace(",", thousand_separator)
        return formatted
    else:
        formatted = f"{int(number):,}"
        if thousand_separator != ",":
            formatted = formatted.replace(",", thousand_separator)
        return formatted
```

```python
@contextmanager
def _db_session(self) -> Generator[Optional[sqlite3.Cursor], None, None]:
    """数据库连接上下文管理器"""
    conn = self._get_db_connection()
    if not conn:
        yield None
        return

    cursor = None
    try:
        cursor = conn.cursor()
        yield cursor
        conn.commit()
    except Exception:
        conn.rollback()
        raise
    finally:

def db_session(self) -> Generator[Optional[sqlite3.Cursor], None, None]:
```
